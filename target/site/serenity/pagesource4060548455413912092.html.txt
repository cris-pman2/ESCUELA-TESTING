<script>
    var sumar = document.getElementById("modal-shopping-add");
    var restar = document.getElementById("modal-shopping-sub");
    var contador = document.getElementById("modal-shopping-quantity");

    function calcular(value) {
        let isValid = /^[0-9][0-9]*$/.test(value);
        if (isValid) {
            let minQuantity = 1;
            let maxQuantity = $("#modal-shopping-cart").data("maxQuantity");
            let modalAlert = $("#modal-alert");

            if (maxQuantity > 0 && value > maxQuantity) {
                sumar.disabled = true;
                modalAlert.show();
            }
            else if (value < minQuantity) {
                restar.disabled = true;
            }
            else {
                contador.value = value;
                sumar.disabled = false;
                restar.disabled = false;
                modalAlert.hide();
                updateCart(value);
            }            
        }
    }

    sumar.onclick = function () {
        value = Number(contador.value) + 1;
        calcular(value);
    };

    restar.onclick = function () {
        value = Number(contador.value) - 1;
        calcular(value);
    };

    function xhrCart(url, data) {
        return $.ajax({
            url: url,
            type: 'POST',
            data: JSON.stringify(data),
            processData: false,
            cache: false,
            contentType: "application/json; charset=utf-8",
            datatype: 'json',
        });
    };

    function updateCart(value) {
        $('#cart__items__preloader').show();

        var seoUrl = $("#modal-shopping-cart").data("id");

        xhrCart('/cart/update/' + seoUrl + '/' + parseInt(value)).done(function (response) {
            if (!response.isSuccess) {
                let errors = [];
                sessionStorage.setItem("hasErrors", "true");
                errors.push(response.message);
                sessionStorage.setItem("errors", JSON.stringify(errors));
            }

            //Get Items Cart
            $.getJSON("/cart/summary", function (data) {
                var total = data.totalItems;
                $('.icon-cart__badge').text(total);
            });

            $('#cart__items__preloader').hide();
        });
    }
