function searchBoxV2(boxesCordsY,eX,eY){
  var maxBottom=0;
  var maxedYL=0;
  var maxedYR=0;
  for(var b=0;b<boxesCordsY.length;b++){
    if(boxesCordsY[b].bottom>maxBottom){
      maxBottom=boxesCordsY[b].bottom;
  }
  var anterior=0;
  var siguiente=eY;
  if(b-1>=0){
      anterior=boxesCordsY[b-1].bottom;
  }
  siguiente=boxesCordsY[b].top;
  createBg(0,anterior,eX,siguiente);
    //izq
    maxedYL=0;
    maxedYR=0;
    for(var bn=0;bn<boxesCordsY.length;bn++){
      if(boxesCordsY[bn].id!=boxesCordsY[b].id){

        var menor = boxesCordsY[b];
        var mayor = boxesCordsY[bn];

        if(boxesCordsY[bn].right<boxesCordsY[b].left&&
          ( boxesCordsY[bn].bottom>boxesCordsY[b].y && boxesCordsY[bn].bottom<boxesCordsY[b].bottom
            ||
            boxesCordsY[bn].y<boxesCordsY[b].bottom && boxesCordsY[bn].y>boxesCordsY[b].y
            )
          ){
          maxedYL=boxesCordsY[bn].bottom;
      if(boxesCordsY[bn].y<boxesCordsY[b].bottom && boxesCordsY[bn].y>boxesCordsY[b].y){
        createBg(
          0,
          boxesCordsY[b].top,
          boxesCordsY[b].left,
          boxesCordsY[bn].top
          );
    } else {
        createBg(
          boxesCordsY[bn].right,
          boxesCordsY[b].top,
          boxesCordsY[b].left,
          boxesCordsY[bn].bottom
          );
    }
}

if( boxesCordsY[bn].left>boxesCordsY[b].right&&

  maxedYR=boxesCordsY[bn].bottom;

      topeAltura=menor.bottom
  } else {
      topeAltura=mayor.bottom;
  }
  createBg(
      boxesCordsY[b].right,

      boxesCordsY[bn].left,
      topeAltura,
      "right"
      );
}
}
}
}
if(maxedYL==0){
  createBg(
    0,

function deleteOldBg(query){
  var antiguosBg = document.querySelectorAll(query);
  for(var a=0;a<antiguosBg.length;a++){
    antiguosBg[a].parentElement.removeChild(antiguosBg[a]);
}
}


function generateBg(boxesQuerys){
  deleteOldBg(".backgroundForTour");
  if(boxesQuerys.length){
    var boxes=[];
    for(var b=0;b<boxesQuerys.length;b++){
      boxes.push(document.querySelector(boxesQuerys[b]));
  }
  deleteOldBg(".tour-backdrop");
  var height = window.innerHeight;
  var width = window.innerWidth;
  var boxesCordsX=[];
  var boxesCordsY=[];
  for(var b=0;b<boxes.length;b++){
      var o = boxes[b].getBoundingClientRect();
      o.id=b;
      boxesCordsX.push(o);
      boxesCordsY.push(o);
  }
  boxesCordsY.sort(function(a, b) {
    return a.y - b.y;
});
  searchBoxV2(boxesCordsY,width,height);
}
}
